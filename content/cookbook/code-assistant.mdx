# Code Assistant

Build an AI coding assistant that can write, run, and debug code.

## What We're Building

A coding assistant that:
- Writes code in multiple languages
- Executes code safely in a sandbox
- Debugs and explains errors
- Suggests improvements

## Quick Start

```python
from isa_agent_sdk import Agent

agent = Agent(
    name="code-assistant",
    model="claude-sonnet-4-20250514",
    tools=["code_interpreter", "file_read", "file_write"],
    system_prompt="You are an expert programmer. Write clean, efficient code."
)

response = await agent.run("Write a function to find prime numbers up to n")
```

```python
from isa_agent_sdk import Agent

class CodeAssistant:
    def __init__(self):
        self.agent = Agent(
            name="coder",
            model="claude-sonnet-4-20250514",
            tools=[
                "code_interpreter",  # Execute Python code
                "shell",             # Run shell commands
                "file_read",         # Read files
                "file_write",        # Write files
            ],
            system_prompt="""You are an expert software engineer.

            When writing code:
            - Write clean, well-documented code
            - Follow best practices and conventions
            - Include error handling
            - Add type hints (Python) or types (TypeScript)

            When debugging:
            - Analyze the error carefully
            - Explain what went wrong
            - Provide a fix with explanation
            """
        )

    async def write_code(self, task: str) -> str:
        response = await self.agent.run(f"Write code to: {task}")
        return response.content

    async def debug_code(self, code: str, error: str) -> str:
        prompt = f"""Debug this code:

```
{code}
```

Error: {error}
"""
        response = await self.agent.run(prompt)
        return response.content

    async def run_code(self, code: str) -> str:
        prompt = f"Execute this code and show the output:\n```python\n{code}\n```"
        response = await self.agent.run(prompt)
        return response.content

# Usage
assistant = CodeAssistant()

# Write code
code = await assistant.write_code("Sort a list of dictionaries by a specific key")

# Run code
result = await assistant.run_code("""
def fibonacci(n):
    a, b = 0, 1
    for _ in range(n):
        yield a
        a, b = b, a + b

print(list(fibonacci(10)))
""")
```

```javascript
import { Agent } from 'isa-agent-sdk';

class CodeAssistant {
    constructor() {
        this.agent = new Agent({
            name: "coder",
            model: "claude-sonnet-4-20250514",
            tools: ["code_interpreter", "shell", "file_read", "file_write"],
            systemPrompt: `You are an expert software engineer...`
        });
    }

    async writeCode(task) {
        const response = await this.agent.run(`Write code to: ${task}`);
        return response.content;
    }

    async debugCode(code, error) {
        const response = await this.agent.run(
            `Debug this code:\n\`\`\`\n${code}\n\`\`\`\n\nError: ${error}`
        );
        return response.content;
    }
}

const assistant = new CodeAssistant();
const code = await assistant.writeCode("Implement a binary search tree");
```

## Safe Code Execution

The `code_interpreter` tool runs code in a sandboxed environment:

```python
from isa_agent_sdk import Agent
from isa_agent_sdk.tools import CodeInterpreter

# Configure the sandbox
interpreter = CodeInterpreter(
    timeout=30,           # Max execution time (seconds)
    memory_limit="512MB", # Memory limit
    allowed_modules=[     # Whitelist of modules
        "math", "json", "datetime", "collections",
        "numpy", "pandas"  # Data science libraries
    ]
)

agent = Agent(
    name="safe-coder",
    tools=[interpreter]
)
```
